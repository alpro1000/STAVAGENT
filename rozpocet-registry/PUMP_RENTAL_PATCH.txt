// Add to imports (line ~30):
import { getSuppliers, getSupplier, calculateArrival, calculateOperation } from '../../services/pumpCalculator';

// Add state after line ~70 (after isExpanded):
const [selectedSupplierId, setSelectedSupplierId] = useState<string | undefined>(data.supplier_id);

// Update selectPumpType function (line ~90):
const selectPumpType = (supplierId: string, pumpName: string) => {
  if (!supplierId || !pumpName) {
    update({ supplier_id: undefined, supplier_name: undefined, pump_type_id: undefined, pump_label: undefined });
    return;
  }
  const supplier = getSupplier(supplierId);
  if (!supplier) return;
  
  const pump = supplier.pumps.find(p => p.name === pumpName);
  if (!pump) return;
  
  // Auto-fill parameters based on supplier billing model
  const arrival = calculateArrival(supplier, pump, data.vzdalenost_km || 10);
  const operation = calculateOperation(supplier, pump, 1, 1); // 1h, 1m³ for rate extraction
  
  update({
    supplier_id: supplier.id,
    supplier_name: supplier.name,
    pump_type_id: pumpName,
    pump_label: pumpName,
    manipulace_czk_h: pump.operation_per_h || operation,
    priplatek_czk_m3: pump.volume_per_m3 || 0,
    pristaveni_fixed_czk: pump.arrival_fixed || 0,
    czk_km: pump.arrival_per_km || 0,
    vykon_m3h: 30, // Default, user can override
  });
};

// Update computeTotals function (line ~120):
function computeTotals(d: PumpRentalData): PumpRentalData {
  const items = d.items.map(i =>
    recomputeItem(i, d.vykon_m3h, d.stavba_h, d.myti_h),
  );

  const celkem_m3 = items.reduce((s, i) => s + i.celkem_m3, 0);
  const celkem_pristaveni = items.reduce((s, i) => s + i.pocet_pristaveni, 0);
  const celkem_hodiny = items.reduce((s, i) => s + i.hodiny_celkem, 0);

  // Use supplier-specific calculation if available
  let celkem_doprava = 0;
  let celkem_manipulace = 0;
  let celkem_priplatek_m3 = 0;

  if (d.supplier_id && d.pump_type_id) {
    const supplier = getSupplier(d.supplier_id);
    if (supplier) {
      const pump = supplier.pumps.find(p => p.name === d.pump_type_id);
      if (pump) {
        celkem_doprava = calculateArrival(supplier, pump, d.vzdalenost_km) * celkem_pristaveni;
        celkem_manipulace = calculateOperation(supplier, pump, celkem_hodiny, celkem_m3);
        celkem_priplatek_m3 = 0; // Already included in operation for hourly_plus_m3
      }
    }
  }

  // Fallback to legacy calculation
  if (celkem_doprava === 0) {
    celkem_doprava = celkem_pristaveni * (d.pristaveni_fixed_czk + d.vzdalenost_km * d.czk_km * 2);
    celkem_manipulace = d.manipulace_czk_h * celkem_hodiny;
    celkem_priplatek_m3 = d.priplatek_czk_m3 * celkem_m3;
  }

  const accessories = d.accessories.map(a => ({
    ...a,
    celkem: a.mnozstvi * a.czk_per_unit,
  }));
  const celkem_prislusenstvi = accessories.reduce((s, a) => s + a.celkem, 0);

  const surcharges = d.surcharges.map(s => ({
    ...s,
    celkem: s.czk_per_pristaveni * celkem_pristaveni,
  }));
  const celkem_priplatky = surcharges.reduce((s, x) => s + x.celkem, 0);

  const konecna_cena =
    celkem_doprava +
    celkem_manipulace +
    celkem_priplatek_m3 +
    celkem_prislusenstvi +
    celkem_priplatky;

  return {
    ...d,
    items,
    accessories,
    surcharges,
    celkem_m3,
    celkem_pristaveni,
    celkem_hodiny,
    celkem_doprava,
    celkem_manipulace,
    celkem_priplatek_m3,
    celkem_prislusenstvi,
    celkem_priplatky,
    konecna_cena,
  };
}
